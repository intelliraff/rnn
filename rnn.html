import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import MinMaxScaler
from keras.models import Sequential
from keras.layers import SimpleRNN, Dense, Dropout

# Load dataset
dataset = pd.read_csv('pollution_actual.csv', nrows=1000)

# Set exact target column
target_column = 'pollution'

# Extract and clean target column
data = dataset[[target_column]].dropna()
values = data.values

# Scale data
sc = MinMaxScaler(feature_range=(0, 1))
scaled_values = sc.fit_transform(values)

# Prepare training data
x_train, y_train = [], []
n_past = 30
n_future = 1

for i in range(n_past, len(scaled_values) - n_future + 1):
    x_train.append(scaled_values[i - n_past:i, 0])
    y_train.append(scaled_values[i + n_future - 1:i + n_future, 0])

x_train, y_train = np.array(x_train), np.array(y_train)
x_train = np.reshape(x_train, (x_train.shape[0], x_train.shape[1], 1))

# Build Simple RNN model
model = Sequential()
model.add(SimpleRNN(units=50, return_sequences=True, input_shape=(x_train.shape[1], 1)))
model.add(Dropout(0.2))
model.add(SimpleRNN(units=50))
model.add(Dropout(0.2))
model.add(Dense(units=1, activation='linear'))

model.compile(optimizer='adam', loss='mean_squared_error')

# Train model
history = model.fit(x_train, y_train, epochs=50, batch_size=32, verbose=1)

# Make prediction on last available window
test_input = scaled_values[-n_past:]
test_input = np.reshape(test_input, (1, n_past, 1))

predicted_value = model.predict(test_input)
predicted_value = sc.inverse_transform(predicted_value)
print(f"Predicted next value (Simple RNN): {predicted_value.flatten()[0]}")
